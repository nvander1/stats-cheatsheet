\subsection{Sample Mean and Sample Variance}
\begin{defi}{Sample Mean}
$$\bar{x} = \sum_{i=1}^n x_i/n$$
\end{defi}

$$y_i = ax_i + b \Rightarrow \bar{y} = a \bar{x} + b$$

Given a frequency table with values $v_i$ and frequencies $f_i$:
$$\bar{x} = \sum_{i=1}^k v_if_i/n$$

\begin{defi}{Sample Variance}
$$s^2 = \left(\sum_{i=1}^n x_i^2 - n\bar{x}^2\right)/(n-1)$$
\end{defi}

$$y_i = ax_i + b \Rightarrow s_y^2 = as_x^2$$

\subsection{Sample Percentiles}
With data $x$ in increasing order:
\begin{equation*}
100p \text{ Percentile} = \begin{cases}
\lceil x_{np}\rceil &,\text{$np$ is not an integer}\\
\frac{1}{2}x_{np}x_{np+1} &, \text{otherwise}
\end{cases}
\end{equation}

\begin{defi}{Sample Correlation Coefficient}
$$\frac{1}{(n-1)s_xs_y}\sum_{i=1}^n(x_i-\bar{x})(y_i-\bar{y})$$
\end{defi}
